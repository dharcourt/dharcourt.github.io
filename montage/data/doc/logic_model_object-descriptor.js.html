var bootstrap = require("logic/bootstrap/bootstrap.js"),
    Montage = require("montage").Montage,
    PropertyDescriptor = require("logic/model/property-descriptor").PropertyDescriptor,
    RelationshipDescriptor = require("logic/model/relationship-descriptor").RelationshipDescriptor;

/**
 * Describes a type of object.
 *
 * Because of [an issue]{@linkplain https://github.com/jsdoc3/jsdoc/issues/1049}
 * with JSDoc the `prototype` property of this class will not appear in the
 * JSDoc generated documentation for this class. This property does exist,
 * however. It is of type `Object` and is a reference to the prototype to use
 * for objects described by instances of this class.
 *
 * @class
 * @extends external:Montage
 */
exports.ObjectDescriptor = Montage.specialize(/** @lends ObjectDescriptor# */{

    /**
     * The name of the type of object described by this descriptor.
     *
     * @type {string}
     */
    name: {
        value: undefined
    },

    /**
     * The prototype to use for objects of this type.
     *
     * @type {Object}
     */
    prototype: {
        get: function () {
            if (!this._prototype) {
                this._prototype = new Montage();
            }
            return this._prototype;
        },
        set: function (prototype) {
            prototype = prototype || new Montage();
            if (prototype !== this._prototype && this._prototype) {
                this._removeTriggers(Object.keys(this.properties));
            }
            if (prototype !== this._prototype) {
                this._prototype = prototype;
                this._addTriggers(Object.keys(this.properties));
            }
        }
    },

    /**
     * Descriptors of the properties of this type of object, by property name.
     *
     * The object provided by this property should not be modified by any object
     * other than {@link ObjectDescriptor}.
     *
     * @type {Object.<string, PropertyDescriptor>}
     */
    properties: {
        value: {}
    },

    /**
     * Convenience method to define a relationship without having to manually
     * create a {@link RelationshipDescriptor} or one or more
     * [PropertyDescriptors]{@link PropertyDescriptor}.
     *
     * @method
     * @argument {Object} definition - An object defining the relationship by
     *                                 providing values for the
     *                                 [destinationType]{@link RelationshipDescriptor#destinationType},
     *                                 [valueExpressions]{@link RelationshipDescriptor#valueExpression},
     *                                 and
     *                                 [criteriaExpressions]{@link RelationshipDescriptor#criteriaExpressions}
     *                                 properties in the
     *                                 {@link RelationshipDescriptor} that will
     *                                 be created by this method.
     */
    addRelationship: {
        value: function (definition) {
            var relationship = new RelationshipDescriptor(),
                keys = Object.keys(definition.valueExpressions),
                property, i, n;
            relationship.destinationType = definition.destinationType;
            relationship.valueExpressions = definition.valueExpressions;
            relationship.criteriaExpressions = definition.criteriaExpressions;
            for (i = 0, n = keys.length; i < n; ++i) {
                property = new PropertyDescriptor();
                property.relationship = relationship;
                this.properties[keys[i]] = property;
                this._addTrigger(keys[i]);
            }
        }
    },

    /**
     * @private
     * @method
     */
     _addTriggers: {
        value: function (names) {
            var property, i, n;
            for (i = 0, n = names.length; i < n; ++i) {
                property = this.properties[names[i]];
                if (property && property.relationship) {
                    this._addTrigger(names[i]);
                }
            }
        }
     },

    /**
     * @private
     * @method
     */
     _addTrigger: {
        value: function (name, property) {
            var self = this,
                replaced = Object.getOwnPropertyDescriptor(this.prototype, name),
                replacement = {
                    get: function () {
                        self._triggerGet.call(self, this, name);
                    },
                    set: function (value) {
                        self._triggerSet.call(self, this, name, value);
                    },
                    enumerable: true
                };
            Montage.defineProperty(this.prototype, name, replacement);
            this._triggerReplaceds = this._triggerReplaceds || new WeakMap();
            this._triggerReplaceds.set(replacement.get, replaced);
        }
     },

    /**
     * @private
     * @method
     */
     _removeTriggers: {
        value: function (names) {
            var i, n;
            for (i = 0, n = names.length; i < n; ++i) {
                this._removeTrigger(names[i]);
            }
        }
     },

    /**
     * @private
     * @method
     */
     _removeTrigger: {
        value: function (name) {
            var replacement = Object.getOwnPropertyDescriptor(this.prototype, name);
                replaced = replacement && this._triggerReplaceds && this._triggerReplaceds.get(replacement.get);
            if (replaced) {
                Montage.defineProperty(this.prototype, name, replaced);
                this._triggerReplaceds.delete(replacement.get);
            }
        }
     },

    /**
     * @private
     * @method
     */
     _triggerGet: {
        value: function (object, name) {
            // Let a service fetch this property's value asynchronously.
            DataService.rootService._getProperty(object, name);
            // Now that the value is being fetched, remove the trigger to
            // prevent multiple fetches.
            this._removeTrigger(name);
            // Now that the trigger is removed the following is sufficient to
            // get the property's current value (usually undefined).
            return object[name];
        }
     },

    /**
     * @private
     * @method
     */
     _triggerSet: {
        value: function (object, name, value) {
            // Remove this trigger since we don't want the value set here to be
            // later overridden by a triggerred fetch.
            this._removeTrigger(name);
            // Now that the trigger is removed the following is sufficient to
            // set the property's value.
            object[name] = value;
        }
     }

});
